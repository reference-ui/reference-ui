import { Button, Div } from '@reference-ui/react'

# Get Started

## Installation

```bash
pnpm add @reference-ui/react
```

## Usage

Import the components and styles you need:

```tsx
import { Button, Div } from '@reference-ui/react'
import '@reference-ui/system/styles.css'

function App() {
  return (
    <Div>
      <Button>Click me</Button>
    </Div>
  )
}
```

## Try it

Here's a live example:

  <Button>Design system button</Button>
</Div>
